void app_init(const void *unused)
{
  UNUSED_PARAMETER(unused);
  osThreadNew((osThreadFunc_t)application_start, NULL, &thread_attributes);
}

#if LOAD_CERTIFICATE
sl_status_t clear_and_load_certificates_in_flash(void)
{
  sl_status_t status;

  status = sl_net_set_credential(SL_NET_TLS_SERVER_CREDENTIAL_ID(0),
                                 SL_NET_SIGNING_CERTIFICATE,
                                 silabs_dgcert_ca,
                                 (sizeof(silabs_dgcert_ca) - 1));
  if (status != SL_STATUS_OK) {
    printf("\r\nLoading TLS CA certificate in to FLASH Failed, Error Code : 0x%lX\r\n", status);
  } else {
    printf("\r\nLoad TLS CA certificate at index %d Success\r\n", CERTIFICATE_INDEX);
  }
  return status;
}
#endif

sl_status_t join_callback_handler(sl_wifi_event_t event, char *result,
                                  uint32_t result_length, void *arg)
{
  UNUSED_PARAMETER(result);
  UNUSED_PARAMETER(arg);
  app_state = WLAN_UNCONNECTED_STATE;
  printf("\r\nIn Join CB\r\n");
  if (SL_WIFI_CHECK_IF_EVENT_FAILED(event)) {
    printf("F: Initiating rejoin %lu bytes payload\n", result_length);
    return SL_STATUS_FAIL;
  }
  return SL_STATUS_OK;
}

// ─────────────────────────────────────────────────────────────────────────────
//  Main application thread — Wi-Fi state machine
// ─────────────────────────────────────────────────────────────────────────────
void application_start(const void *unused)
{
  UNUSED_PARAMETER(unused);
  sl_status_t status = SL_STATUS_OK;

  while (1) {
    switch (app_state) {

      case WLAN_INITIAL_STATE: {
        status = sl_net_init(SL_NET_WIFI_CLIENT_INTERFACE,
                             &station_init_configuration, NULL, NULL);
        if (status != SL_STATUS_OK) {
          printf("\r\nWi-Fi Client initialization failed, Error Code: 0x%lX\r\n", status);
          return;
        }
        printf("\r\nWi-Fi Client initialization success\r\n");

#if LOAD_CERTIFICATE
        status = clear_and_load_certificates_in_flash();
        if (status != SL_STATUS_OK) {
          printf("\r\nUnexpected error while loading certificate: 0x%lX\r\n", status);
          return;
        }
#endif
        // Init GPIO for button (input) and LED (output)
        sl_gpio_driver_init();
        sl_gpio_set_configuration(sl_gpio_pin_config);
        sl_gpio_set_configuration(sl_led_pin_config);

        app_state = WLAN_UNCONNECTED_STATE;
      } break;

      case WLAN_UNCONNECTED_STATE: {
        sl_wifi_set_join_callback(join_callback_handler, NULL);
        status = sl_net_up(SL_NET_WIFI_CLIENT_INTERFACE,
                           SL_NET_DEFAULT_WIFI_CLIENT_PROFILE_ID);
        if (status != SL_STATUS_OK) {
          printf("\r\nFailed to bring Wi-Fi client interface up: 0x%lX\r\n", status);
          app_state = WLAN_UNCONNECTED_STATE;
          break;
        }
        printf("\r\nWi-Fi Client interface up\r\n");
        app_state = WLAN_FIRMWARE_UPDATE;   // transition to "ready" state
      } break;

      case WLAN_FIRMWARE_UPDATE: {
        // ── OTA disabled: spin up MQTT thread instead ─────────────────────
        printf("\r\nWi-Fi connected. Starting MQTT thread...\r\n");
        osThreadNew(mqtt_thread_entry, NULL, &mqtt_thread_attributes);
        app_state = WLAN_OTA_UPDATE_DONE;   // park main thread
      } break;

      case WLAN_OTA_UPDATE_DONE:
      default:
        osDelay(osWaitForever); // idle — MQTT thread takes over
        break;
    }
  }
}

// ─────────────────────────────────────────────────────────────────────────────
//  MQTT event callback
// ─────────────────────────────────────────────────────────────────────────────
static void mqtt_event_handler(void *client, sl_mqtt_client_event_t event,
                               void *event_data, void *context)
{
  (void)event_data;
  (void)context;
  sl_status_t status;

  switch (event) {

    case SL_MQTT_CLIENT_CONNECTED_EVENT:
      printf("\r\n[MQTT] Connected to broker\r\n");
      mqtt_connected = 1;

      // Subscribe to LED control topic from Node-RED
      status = sl_mqtt_client_subscribe(client,
                                        (uint8_t *)LED_CONTROL_TOPIC,
                                        strlen(LED_CONTROL_TOPIC),
                                        MQTT_QOS,
                                        0,
                                        mqtt_message_handler,
                                        (void *)LED_CONTROL_TOPIC);
      if (status != SL_STATUS_IN_PROGRESS) {
        printf("[MQTT] Subscribe failed: 0x%lX\r\n", status);
      }
      break;

    case SL_MQTT_CLIENT_SUBSCRIBED_EVENT:
      printf("[MQTT] Subscribed to: %s\r\n", LED_CONTROL_TOPIC);
      break;

    case SL_MQTT_CLIENT_MESSAGE_PUBLISHED_EVENT:
      // publish ACK — nothing to do
      break;

    case SL_MQTT_CLIENT_DISCONNECTED_EVENT:
      printf("[MQTT] Disconnected\r\n");
      mqtt_connected = 0;
      mqtt_done      = 1;
      break;

    case SL_MQTT_CLIENT_ERROR_EVENT:
      printf("[MQTT] Error event\r\n");
      mqtt_connected = 0;
      mqtt_done      = 1;
      break;

    default:
      break;
  }
}

// ─────────────────────────────────────────────────────────────────────────────
//  MQTT message callback — Node-RED LED switch → MCU LED
//  Expected payload: "ON" = LED on, "OFF" = LED off
// ─────────────────────────────────────────────────────────────────────────────
static void mqtt_message_handler(void *client, sl_mqtt_client_message_t *msg,
                                 void *context)
{
  (void)client;
  (void)context;

  if (msg->content_length >= 2 &&
      strncmp((char *)msg->content, "ON", msg->content_length) == 0) {
    sl_gpio_driver_set_pin(&sl_led_pin_config.port_pin);
    printf("[MQTT] LED ON\r\n");
  } else if (msg->content_length >= 3 &&
             strncmp((char *)msg->content, "OFF", msg->content_length) == 0) {
    sl_gpio_driver_clear_pin(&sl_led_pin_config.port_pin);
    printf("[MQTT] LED OFF\r\n");
  }
}

// ─────────────────────────────────────────────────────────────────────────────
//  MQTT thread — connects to broker, polls button, publishes state
// ─────────────────────────────────────────────────────────────────────────────
static void mqtt_thread_entry(void *argument)
{
  (void)argument;
  sl_status_t status;

  // ── 1. Configure broker ──────────────────────────────────────────────────
  mqtt_broker.port                    = MQTT_BROKER_PORT;
  mqtt_broker.is_connection_encrypted = 0;
  mqtt_broker.connect_timeout         = MQTT_CONNECT_TIMEOUT;
  mqtt_broker.keep_alive_interval     = MQTT_KEEP_ALIVE;
  mqtt_broker.keep_alive_retries      = MQTT_KEEPALIVE_RETRIES;

  status = sl_net_inet_addr(MQTT_BROKER_IP, &mqtt_broker.ip.ip.v4.value);
  if (status != SL_STATUS_OK) {
    printf("[MQTT] IP conversion failed: 0x%lX\r\n", status);
    return;
  }
  mqtt_broker.ip.type = SL_IPV4;

  // ── 2. Configure client ──────────────────────────────────────────────────
  mqtt_cfg.is_clean_session = 1;
  mqtt_cfg.client_id        = (uint8_t *)MQTT_CLIENT_ID;
  mqtt_cfg.client_id_length = strlen(MQTT_CLIENT_ID);
  mqtt_cfg.client_port      = MQTT_CLIENT_PORT;

  // ── 3. Init MQTT client ──────────────────────────────────────────────────
  status = sl_mqtt_client_init(&mqtt_client, mqtt_event_handler);
  if (status != SL_STATUS_OK) {
    printf("[MQTT] Init failed: 0x%lX\r\n", status);
    return;
  }
  printf("[MQTT] Client init success\r\n");

  // ── 4. Connect ───────────────────────────────────────────────────────────
  status = sl_mqtt_client_connect(&mqtt_client, &mqtt_broker,
                                  NULL,           // no last-will
                                  &mqtt_cfg, 0);
  if (status != SL_STATUS_IN_PROGRESS) {
    printf("[MQTT] Connect failed: 0x%lX\r\n", status);
    return;
  }

  // Wait for connection
  while (!mqtt_connected && !mqtt_done) {
    osThreadYield();
  }
  if (!mqtt_connected) {
    printf("[MQTT] Could not connect to broker\r\n");
    return;
  }

  // ── 5. Button poll loop — publish state changes ──────────────────────────
  uint8_t prev_pin = 0xFF; // force first publish

  while (!mqtt_done) {
    uint8_t pin_value = 1;
    sl_gpio_driver_get_pin(&sl_gpio_pin_config.port_pin, &pin_value);

    // Button is active-low (pressed = 0)
    uint8_t btn_pressed = (pin_value == 0) ? 1 : 0;

    if (btn_pressed != prev_pin) { // only publish on state change
      prev_pin = btn_pressed;

      const char *payload = btn_pressed ? "ON" : "OFF";

      sl_mqtt_client_message_t pub_msg = {
        .qos_level            = MQTT_QOS,
        .is_retained          = 0,
        .is_duplicate_message = 0,
        .topic                = (uint8_t *)BUTTON_STATE_TOPIC,
        .topic_length         = strlen(BUTTON_STATE_TOPIC),
        .content              = (uint8_t *)payload,
        .content_length       = strlen(payload),
      };

      status = sl_mqtt_client_publish(&mqtt_client, &pub_msg, 0, NULL);
      if (status != SL_STATUS_IN_PROGRESS) {
        printf("[MQTT] Publish failed: 0x%lX\r\n", status);
      } else {
        printf("[MQTT] Published button state: %s\r\n", payload);
      }
    }

    osDelay(BUTTON_POLL_MS);
  }

  printf("[MQTT] Thread exiting\r\n");
}
